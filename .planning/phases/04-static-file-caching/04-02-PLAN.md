```python
def get_track_geometry_cache() -> CachedData[list[dict[str, Any]]]:
    """Create cached track geometry instance with TTL refresh.

    Returns:
        CachedData instance configured for track geometry with 30-day TTL

    Raises:
        ValueError: If cache configuration missing from config
    """
    cfg = load_config()
    cache_config = cfg.get("cache", {})
    cache_dir = cache_config.get("directory", ".cache")
    ttl = int(cache_config.get("track_geometry_ttl", 2592000))  # 30 days default

    cache_file = Path(cache_dir) / "cta_track_geometry.json"

    def fetch_fn() -> list[dict[str, Any]]:
        """Fetch and normalize track geometry from Chicago Data Portal."""
        with httpx.Client(timeout=30.0) as client:
            raw_data = get_cta_track_shapes(client)
            return normalize_cta_track_shapes(raw_data)

    return CachedData(cache_file=cache_file, ttl=ttl, fetch_fn=fetch_fn)
```
